    # volcano-ui

    # Install

    npm i volcano-ui -S

    # Quick Start

    在 main.js 中写入以下内容：

    import Vue from "vue";
import VolcanoUI from "volcano-ui";
import "volcano-ui/lib/volcano-ui.css";
import App from "./App.vue";
Vue.use(VolcanoUI);

    # On demand

    如果只希望引入部分组件，那么需要在 main.js 中写入以下内容：

    import Vue from "vue";
import { reCard } from "volcano-ui";
import App from "./App.vue";
Vue.use(reCard);
/* or
 *
 * Vue.component(reTable.name, reTable);
 * Vue.component(reCard.name, reCard);
 */

new Vue({
  el: "#app",
  render: (h) => h(App),
});
